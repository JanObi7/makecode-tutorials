let licht = 0
input.onSound(DetectedSound.Loud, function () {
    if (licht == 3) {
        licht = 2

            # # # # #
            . # # # .
            . . # . .
            `)
    } else if (licht == 2) {
        licht = 1
        basic.showLeds(`
            . . . . .
            . . # . .
            . # # # .
            . . # . .
            . . . . .
            `)
    } else if (licht == 1) {
        licht = 0
        basic.showLeds(`
            . . . . .
            . . . . .
            . . . . .
            . . . . .
            . . . . .
            `)
    } else {
        licht = 3
        basic.showLeds(`
            . # # # .
            # # # # #
            # # # # #
            # # # # #
            . # # # .
            `)
    }
})
